export const db = globalForPrisma.prisma ??
  new PrismaClient({
    log: process.env.NODE_ENV === 'development' ? ['query', 'error', 'warn'] : ['error'],
    datasources: {
      db: {
        url: process.env.DATABASE_URL,
      },
    },

    pool_timeout: 10,
  });

This will:
- Limit to 10 connections max (currently using 4)
- Set 10 second timeout for connection acquisition
- Save 10-90MB of memory (from potential 100 connections to max 10)
